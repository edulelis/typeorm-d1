import { DataSource } from "typeorm/data-source/DataSource";
import { AbstractSqliteDriver } from "typeorm/driver/sqlite-abstract/AbstractSqliteDriver";
import { D1QueryRunner } from "./D1QueryRunner";
import { D1Database } from "./types";
import { ReplicationMode } from "typeorm/driver/types/ReplicationMode";
import { QueryRunner } from "typeorm/query-runner/QueryRunner";
import { ColumnType } from "typeorm/driver/types/ColumnTypes";
 
/**
 * D1Driver is the main driver class that connects TypeORM to Cloudflare D1
 * It extends AbstractSqliteDriver since D1 is SQLite-based
 */
export class D1Driver extends AbstractSqliteDriver {
  /**
   * D1 database instance
   * This is set in createDatabaseConnection()
   */
  declare databaseConnection: D1Database;
 
  /**
   * Connection options with D1-specific properties
   */
  declare options: any;
 
  /**
   * Constructor takes a DataSource (connection) as required by TypeORM 0.3
   */
  constructor(connection: DataSource) {
    super(connection);
    this.connection = connection;
    this.options = connection.options;
    
    // Get D1Database from options.driver.database
    // This is set when creating DataSource with driver options
    const driverOptions = (this.options as any).driver as any;
    if (!driverOptions?.database) {
      throw new Error("D1 database instance must be provided in driver options. Use { driver: { database: env.DB } }");
    }
    
    // Extend supported data types to include timestamp (mapped to TEXT in SQLite)
    // The parent class initializes supportedDataTypes, so we add to it
    Eif (this.supportedDataTypes && !this.supportedDataTypes.includes("timestamp")) {
      this.supportedDataTypes.push("timestamp");
    }
  }
 
  /**
   * Creates connection with the database.
   * For D1, we just return the D1Database instance that was passed in options
   * This is called by the base class during connect() and the result is assigned to databaseConnection
   */
  protected async createDatabaseConnection(): Promise<D1Database> {
    const driverOptions = (this.options as any).driver as any;
    const database = driverOptions?.database as D1Database;
    
    if (!database || typeof database.prepare !== "function") {
      throw new Error("Invalid D1 database instance. Expected D1Database with prepare() method.");
    }
    
    return database;
  }
 
  /**
   * Closes connection with database.
   * For D1, there's no persistent connection to close, so this is a no-op
   */
  async disconnect(): Promise<void> {
    this.queryRunner = undefined;
    this.databaseConnection = undefined as any;
  }
 
  /**
   * Creates a query runner used to execute database queries.
   * For D1, we create a new query runner for each request to avoid transaction state conflicts
   * in concurrent scenarios.
   */
  createQueryRunner(mode: ReplicationMode): QueryRunner {
    // Always create a new query runner to avoid transaction state conflicts
    // This is safe for D1 since there's no connection pooling overhead
    return new D1QueryRunner(this);
  }
 
  /**
   * Makes any action after connection (e.g. create extensions in Postgres driver).
   * For D1, we enable foreign key constraints which are disabled by default in SQLite
   */
  async afterConnect(): Promise<void> {
    // Enable foreign key constraints for SQLite/D1
    // This is important for enforcing referential integrity
    const queryRunner = this.createQueryRunner("master");
    try {
      await queryRunner.query("PRAGMA foreign_keys = ON");
    } finally {
      await queryRunner.release();
    }
  }
 
  /**
   * Normalizes type definition for D1/SQLite
   */
  normalizeType(column: {
    type?: ColumnType;
    length?: number | string;
    precision?: number | null;
    scale?: number;
  }): string {
    // For D1/Edge runtime, Buffer is not available, so we skip Buffer-specific handling
    // The base class will handle blob types via the column type string
    return super.normalizeType(column);
  }
 
  /**
   * For SQLite/D1, we don't support multiple databases in the traditional sense
   * This returns the table name as-is
   */
  buildTableName(tableName: string, _schema?: string, database?: string): string {
    // D1 doesn't support schemas or multiple databases
    return tableName;
  }
 
  /**
   * Transaction support for D1
   * D1 supports simple transactions via batch API
   * Override the base class property
   */
  transactionSupport: "simple" | "nested" | "none" = "simple";
}
